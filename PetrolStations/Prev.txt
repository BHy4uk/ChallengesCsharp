using System;
using System.Linq;

namespace PetrolStations
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("Enter coordinate of city A");
            int nA = Convert.ToInt32(Console.ReadLine());
            int c_A = nA;
            Console.WriteLine("Enter coordinate of city B");
            int nB = Convert.ToInt32(Console.ReadLine());
            int c_B = nB;
            int citiesDist = c_B - c_A;

            Console.WriteLine("Enter quantity of miles k that we can drive without refilling");
            int k = Convert.ToInt32(Console.ReadLine());
            int fullTank_km = k;
            int tank = fullTank_km;

            Console.WriteLine("Enter quantity n of the petrol stations");
            int tankQuantity = Convert.ToInt32(Console.ReadLine());

            Console.WriteLine("Coordinate s_i for each tank station");
            var s_i = Console.ReadLine().Split().Select(int.Parse).ToArray();
            Array.Sort(s_i);

            int count = 0;
            int dist;

            for (int i = 0; i < s_i.Length; i++)
            {

                if (i == 0) dist = s_i[i] - c_A;
                else if (i == s_i.Length - 1) dist = citiesDist - s_i[i];
                else dist = s_i[i + 1] - s_i[i];

                if (dist > fullTank_km)
                {
                    Console.WriteLine("-1");

                    break;
                }

                tank -= dist;
                if (tank == 0)
                {
                    count++;
                    tank = fullTank_km;

                }
                else if (tank < 0)
                {
                    count++;
                    tank = fullTank_km;


                }
                if (i == s_i.Length - 1) 
                 Console.WriteLine("{0}" , count);
            }          
        }
    }
}
